Bonus: Improved normalizeMediaUrl (recommended)
While you’re grabbing the logs, replace your current function with this more forgiving version (it handles the most common mismatches and adds warnings):
export function normalizeMediaUrl(url: string | null | undefined): string | null {
  if (!url) return null;
  let clean = url.trim();

  // Already perfect
  if (clean.startsWith("/objects/uploads/")) return clean;
  if (clean.startsWith("objects/uploads/")) return `/${clean}`;

  // Legacy formats
  if (clean.startsWith("/uploads/")) return `/objects${clean}`;
  if (clean.startsWith("uploads/")) return `/objects/${clean}`;

  // Any full external or special URL
  if (/^(https?:\/\/|blob:|data:)/i.test(clean)) return clean;

  // Anything else that starts with / — keep it but warn
  if (clean.startsWith("/")) {
    console.warn("⚠️ Potentially malformed media URL (will be used as-is):", clean);
    return clean;
  }

  console.warn("❌ Unrecognized media URL format → null:", clean);
  return null;
}
